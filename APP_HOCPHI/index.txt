        function apiCall(payload) {
            return new Promise(function(resolve, reject) {
                google.script.run

            return new Promise(function(resolve, reject) {
                google.script.run

                    .withFailureHandler(reject)
                    .loadInitialData();
            });

        async function loadData() {
            try {
                document.getElementById('loading').style.display = 'block';
                document.getElementById('app-content').style.display = 'none';

                    document.getElementById('loading').innerText = "Lỗi kết nối Excel. Hãy kiểm tra tên các Sheet.";

                allData = data.danhSachHocSinh || [];
                allCaiDat = data.caiDatLop || [];
                allChiPhi = data.danhSachChiPhi || [];

                document.getElementById('app-content').style.display = 'block';

        function updateDashboard(totalStudents, unpaidCount, classRevenue) {
            var selectedMonth = document.getElementById('export-month').value;
            var monthStr = "Tháng " + selectedMonth;
            var globalRevenue = 0;
            if (allData.length > 1) {
                allData.slice(1).forEach(function(s) {
                    if (s[8] !== "Đã nghỉ") {
                        var statusStr = String(s[5]);
                        if (statusStr === "Đã thu") globalRevenue += Number(String(s[4]).replace(/[^0-9.-]+/g,"")) || 0;
                        else if (statusStr.indexOf("Thu ") === 0) globalRevenue += Number(statusStr.replace(/[^0-9]/g, "")) || 0;
                    }
                });
            }

            var totalExpense = 0;
            if (allChiPhi && allChiPhi.length > 1) {
                allChiPhi.slice(1).forEach(function(row) {
                    if (row[3] === monthStr || row[3] === selectedMonth) totalExpense += Number(String(row[2]).replace(/[^0-9.-]+/g, "")) || 0;
                });
            }

            var netProfit = globalRevenue - totalExpense;
            document.getElementById('total-count').innerText = totalStudents;
            document.getElementById('unpaid-count').innerText = unpaidCount;
            document.getElementById('total-revenue').innerText = classRevenue.toLocaleString('vi-VN') + ' đ';
            document.getElementById('global-revenue').innerText = globalRevenue.toLocaleString('vi-VN') + ' đ';
            document.getElementById('total-expense').innerText = totalExpense.toLocaleString('vi-VN') + ' đ';
            document.getElementById('net-profit').innerText = netProfit.toLocaleString('vi-VN') + ' đ';
            drawChart(globalRevenue, totalExpense, netProfit);
        }

            });
        }

        function openDiscountModal() {
            var selectedClass = document.getElementById('class-select').value;
            if (!selectedClass) return alert("Vui lòng chọn lớp học để giảm trừ.");
            var classInfo = allCaiDat.find(function(c) { return c[0] === selectedClass; });
            if (!classInfo || !classInfo[2]) return alert("Lớp này chưa cài đặt số buổi.");
            var feePerSession = Math.round((Number(classInfo[1]) || 0) / (Number(classInfo[2]) || 0));
            document.getElementById('disc-class-name').innerText = selectedClass;
            document.getElementById('disc-session-fee').innerText = feePerSession.toLocaleString('vi-VN') + ' đ';
            document.getElementById('disc-sessions-off').value = '';
            document.getElementById('disc-total-amount').value = '';
            document.getElementById('disc-sessions-off').setAttribute('data-fee', feePerSession);
            document.getElementById('discountModal').style.display = 'flex';
        }

        function closeDiscountModal() { document.getElementById('discountModal').style.display = 'none'; }
        function calculateDiscount() {
            var sessionsOff = Number(document.getElementById('disc-sessions-off').value) || 0;
            var feePerSession = Number(document.getElementById('disc-sessions-off').getAttribute('data-fee')) || 0;

        async function submitDiscount(event) {
            event.preventDefault();
            var selectedClass = document.getElementById('class-select').value;
            var amount = Number(document.getElementById('disc-total-amount').value) || 0;
            if (!confirm("Hệ thống sẽ TRỪ " + amount.toLocaleString('vi-VN') + " đ vào Nợ cũ của lớp " + selectedClass + ". Tiếp tục?")) return;
            var btn = document.getElementById('btn-submit-disc');
            btn.disabled = true; btn.innerText = "Đang xử lý...";
            try {
                var result = await apiCall({ loaiYeuCau: "giamTruDongLoat", lop: selectedClass, soTienGiam: amount });
                if(result && result.thanhCong) { alert("Đã giảm trừ thành công!"); closeDiscountModal(); await loadData(); } 
                else { alert("Lỗi từ server: " + (result ? result.loi : "Không xác định")); }
            } catch(e) { alert("Lỗi hệ thống hoặc Thiếu Sheet!"); } finally { btn.disabled = false; btn.innerText = "Xác nhận giảm"; }
        }

        function confirmPayment() {
            if(!currentPayStudent) return;
            var amountReceived = Number(document.getElementById('pay-amount-received').value) || 0;
            var feeAmount = currentPayStudent.feeAmount; var s = currentStudents[currentPayStudent.index];
            var newStatus = "";
            if (amountReceived === feeAmount && currentPayStudent.noCu === 0) newStatus = "Đã thu";
            else if (amountReceived === 0) newStatus = "Chưa đóng";
            else newStatus = "Thu " + amountReceived.toLocaleString('vi-VN');
            var deltaDebt = feeAmount - amountReceived;
            executePaymentAPI(s[0], s[1], newStatus, currentPayStudent.checkbox, deltaDebt);
            closePaymentModal();
        }

        async function executePaymentAPI(hoTen, lop, newStatus, checkboxElem, deltaDebt) {
            var studentData = allData.find(function(s) { return s[0] === hoTen && s[1] === lop; });
            var oldStatus = studentData ? studentData[5] : "Chưa đóng";
            var oldNoCu = studentData ? Number(studentData[7] || 0) : 0;
            if (studentData) { studentData[5] = newStatus; studentData[7] = oldNoCu + deltaDebt; }
            renderStudents(); 
            if(checkboxElem) checkboxElem.disabled = true;
            try {
                var r1 = await apiCall({ loaiYeuCau: "thuTien", hoTen: hoTen, lop: lop, trangThai: newStatus });
                if(deltaDebt !== 0) await apiCall({ loaiYeuCau: "capNhatNoCu", hoTen: hoTen, lop: lop, tienThua: deltaDebt });
            } catch (error) { alert("Lỗi kết nối khi thanh toán!"); if (studentData) { studentData[5] = oldStatus; studentData[7] = oldNoCu; } renderStudents(); } 
            finally { if(checkboxElem) checkboxElem.disabled = false; }
        }

        function calculateFee() {
            var classInfo = allCaiDat.find(function(c) { return c[0] === document.getElementById('new-student-class').value; });
            var feeInput = document.getElementById('new-student-fee'); var dateInput = document.getElementById('new-student-start-date').value;
            if (!classInfo) return; var fullFee = Number(classInfo[1]) || 0; var totalSessions = Number(classInfo[2]) || 0;
            if (!dateInput || totalSessions === 0) { feeInput.value = fullFee; return; }
            var startDate = new Date(dateInput); var daysInMonth = new Date(startDate.getFullYear(), startDate.getMonth() + 1, 0).getDate();
            var remainingSessions = Math.round(((daysInMonth - startDate.getDate() + 1) / daysInMonth) * totalSessions);
            feeInput.value = Math.round((remainingSessions * (fullFee / totalSessions)) / 1000) * 1000; 
        }
